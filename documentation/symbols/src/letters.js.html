1 'use strict';
  2 /* Функция фильтрации элементов массива использюущая функцию-компаратор передаваемый как параметр func*/
  3 let filter = (func, str) => 
  4 {
  5     let uniquestr = [];
  6     for (let i = 0; i < str.length; i++) 
  7     {
  8         if(func(str[i], i))
  9         {
 10             uniquestr.push(str[i]);
 11         }
 12     }
 13     return uniquestr.join("");
 14 }
 15 
 16 /* функция letters в зависимости от переданных параметров удаляет неуникальные элементы из строки:
 17 * случай true: функция будет искать повторяющиеся символы в строках и удалять их, оставляя на месте только первую встречающуюся букву.
 18 * случай false: функция будет оставлять только последнюю встречающуюся букву.
 19 * случай когда второго параметра нет: функция будет удалять из строки символы, которые встречаются в ней больше одного раза.
 20 */ 
 21 let letters = (input, flag) =>
 22 {
 23     const str = input.split('');
 24     if(flag)
 25     {
 26         return filter((item, index) =>
 27         {
 28             return str.indexOf(item) === index; 
 29         }, str);
 30     }
 31     else if(flag === false)
 32     {
 33         return filter((item, index) =>
 34         {
 35             return str.lastIndexOf(item) === index; 
 36         }, str);
 37     }
 38     else
 39     {
 40         return filter((item) =>
 41         {
 42             return str.lastIndexOf(item) === str.indexOf(item); 
 43         }, str);
 44     }
 45 }
